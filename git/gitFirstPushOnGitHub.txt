// 1. создаем папку на своем компьютере с таким же названием как и проект на GitHub
// в ней в коммандной строке вводим:
git init

// 2. после этого создана пустой проект git
// вносим изменения в поэкт / добавляем какой-то файл после чего вводим
git add .

// 3. коммитим свои зименения на локальном проєкте git
git commit -m "first commit"

// 4. привязываем локальный проект к GitHub
git remote add origin https://github.com/some-login/SomeProject.git

// 5. переносим(копируем) свои изменения на GitHub
git push -u origin master
